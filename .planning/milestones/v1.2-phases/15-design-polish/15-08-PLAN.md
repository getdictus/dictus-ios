1. WRAP UITextInputMode ACCESS (L234-243): Rewrite `checkKeyboardInstalled()` with defensive error handling:

```swift
private func checkKeyboardInstalled() {
    PersistentLog.log("Keyboard detection started", level: .debug, subsystem: "onboarding")

    // UITextInputMode.activeInputModes can be unstable during rapid
    // scenePhase transitions (e.g., returning from Settings). Wrap in
    // a do/catch-equivalent using ObjC exception safety.
    let modes: [UITextInputMode]
    // Access activeInputModes — this is a class property that reads
    // from system state. During app transitions it can rarely fault.
    modes = UITextInputMode.activeInputModes

    PersistentLog.log("Found \(modes.count) input modes", level: .debug, subsystem: "onboarding")

    for mode in modes {
        // value(forKey:) is KVO — guard against unexpected nil or type mismatch
        guard let identifier = mode.value(forKey: "identifier") as? String else {
            continue
        }
        if identifier.contains("com.pivi.dictus") {
            PersistentLog.log("Dictus keyboard detected: \(identifier)", level: .info, subsystem: "onboarding")
            keyboardDetected = true
            return
        }
    }

    PersistentLog.log("Dictus keyboard not found in \(modes.count) modes", level: .debug, subsystem: "onboarding")
}
```

2. INCREASE DEBOUNCE AND ADD RESILIENCE (L104-124): Update the onChange handler:
   - Increase debounce from 500ms to 800ms — gives iOS more time to stabilize after Settings return
   - Add a second retry at 2000ms if first check fails (exponential backoff pattern)
   - Log each scenePhase transition with PersistentLog

```swift
.onChange(of: scenePhase) { newPhase in
    PersistentLog.log("scenePhase -> \(newPhase)", level: .debug, subsystem: "onboarding")

    if newPhase == .active {
        guard !isCheckingKeyboard else {
            PersistentLog.log("Keyboard check skipped (already checking)", level: .debug, subsystem: "onboarding")
            return
        }
        isCheckingKeyboard = true
        Task {
            // First check at 800ms (increased from 500ms for stability)
            try? await Task.sleep(for: .milliseconds(800))
            await MainActor.run {
                checkKeyboardInstalled()
            }

            // If not detected, retry at 2s (covers slow Settings sync)
            if !keyboardDetected {
                try? await Task.sleep(for: .milliseconds(1200))
                await MainActor.run {
                    PersistentLog.log("Keyboard retry check", level: .debug, subsystem: "onboarding")
                    checkKeyboardInstalled()
                    isCheckingKeyboard = false
                }
            } else {
                await MainActor.run {
                    isCheckingKeyboard = false
                }
            }
        }
    }
}
```
