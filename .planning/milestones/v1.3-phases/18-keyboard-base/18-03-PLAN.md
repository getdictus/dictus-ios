    **1. Autocapitalization (updateCapitalization method):**

    Add a method to DictusKeyboardBridge that checks the textDocumentProxy's autocapitalization type and sets the keyboard page accordingly. This must be called:
    - After every character insertion (insertText)
    - After every delete (deleteBackward)
    - After return key
    - When textDidChange fires from KeyboardViewController
    - On initial keyboard appear

    Implementation (adapted from giellakbd-ios verified code):
    ```swift
    func updateCapitalization() {
        guard let proxy = controller?.textDocumentProxy else { return }
        // Don't override caps lock
        guard keyboardView?.page != .capslock else { return }
        // Only autocap on letter pages
        guard keyboardView?.page == .normal || keyboardView?.page == .shifted else { return }

        let autocapType = proxy.autocapitalizationType ?? .sentences

        switch autocapType {
        case .sentences:
            let beforeInput = proxy.documentContextBeforeInput ?? ""
            if beforeInput.isEmpty {
                // Beginning of text field -- capitalize
                keyboardView?.page = .shifted
            } else {
                let trimmed = beforeInput.trimmingCharacters(in: .whitespaces)
                if let lastChar = trimmed.last,
                   ".!?".contains(lastChar),
                   beforeInput.last == " " {
                    // After sentence-ending punctuation + space
                    keyboardView?.page = .shifted
                } else if keyboardView?.page == .shifted {
                    // Was shifted from autocap, now typing -- return to normal
                    keyboardView?.page = .normal
                }
            }
        case .words:
            let beforeInput = proxy.documentContextBeforeInput ?? ""
            if beforeInput.isEmpty || beforeInput.last == " " {
                keyboardView?.page = .shifted
            } else {
                keyboardView?.page = .normal
            }
        case .allCharacters:
            keyboardView?.page = .capslock
        default: // .none
            break
        }
    }
    ```

    Verify the shift handling in DictusKeyboardBridge has these behaviors:
    - Single tap on shift when .normal: set .shifted, set isManualShift = true
    - Single tap on shift when .shifted: set .normal, set isManualShift = false
    - Double tap (within 300ms): set .capslock
    - Single tap on shift when .capslock: set .normal
    - After typing ONE character in .shifted (from manual shift): return to .normal
    - After typing in .capslock: STAY in .capslock (that's the point)

    Track `lastShiftTapTime: Date?` for double-tap detection:
    ```swift
    private var lastShiftTapTime: Date?
    private var isManualShift = false

    func handleShiftTap() {
        let now = Date()
        if let lastTap = lastShiftTapTime,
           now.timeIntervalSince(lastTap) < 0.3 {
            // Double tap -> caps lock
            keyboardView?.page = .capslock
            lastShiftTapTime = nil
            isManualShift = false
        } else {
            // Single tap
            switch keyboardView?.page {
            case .normal:
                keyboardView?.page = .shifted
                isManualShift = true
            case .shifted:
                keyboardView?.page = .normal
                isManualShift = false
            case .capslock:
                keyboardView?.page = .normal
                isManualShift = false
            default:
                break
            }
            lastShiftTapTime = now
        }
        HapticFeedback.keyTapped()
        playSound(.modifier)
    }
    ```
