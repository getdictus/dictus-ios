```swift
public struct FillerWordFilter {
    // Locked word list from CONTEXT.md — do NOT add more
    private static let fillers = ["euh", "hm", "bah", "ben", "voila", "um", "uh", "er"]

    private static let pattern: String = {
        let escaped = fillers.map { NSRegularExpression.escapedPattern(for: $0) }
        // Use \b word boundaries to match whole words only
        return "\\b(" + escaped.joined(separator: "|") + ")\\b"
    }()

    public static func clean(_ text: String) -> String {
        var result = text
        if let regex = try? NSRegularExpression(pattern: pattern, options: .caseInsensitive) {
            result = regex.stringByReplacingMatches(
                in: result,
                range: NSRange(result.startIndex..., in: result),
                withTemplate: ""
            )
        }
        // Collapse multiple spaces
        result = result.replacingOccurrences(of: "  +", with: " ", options: .regularExpression)
        // Remove orphaned punctuation (comma/period preceded only by whitespace or at start)
        result = result.replacingOccurrences(of: "^\\s*[,.]\\s*", with: "", options: .regularExpression)
        result = result.replacingOccurrences(of: "\\s+([,.])", with: "$1", options: .regularExpression)
        return result.trimmingCharacters(in: .whitespaces)
    }
}
```

IMPORTANT edge case: `\b` in regex treats apostrophes as word boundaries in French. Test with "l'humain" to confirm "hm" inside "humain" is NOT matched (it shouldn't be because "hm" != "humain" as a word). If `\b` causes issues with French apostrophe words, switch to lookbehind/lookahead `(?<=\\s|^)` and `(?=\\s|$|[,.])`).
