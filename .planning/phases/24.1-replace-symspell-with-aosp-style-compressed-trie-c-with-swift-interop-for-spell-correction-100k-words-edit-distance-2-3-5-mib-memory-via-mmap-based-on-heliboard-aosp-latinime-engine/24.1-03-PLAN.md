   - Add two-pass lookup before delegating to trie:
   ```swift
   func spellCheck(_ word: String) -> (correction: String, alternatives: [String])? {
       guard !word.isEmpty else { return nil }

       // Two-pass lookup: user dictionary first (learned words are always "correct")
       let lowered = word.lowercased()
       let wordToCheck: String
       if let apoIndex = lowered.lastIndex(of: "'") {
           wordToCheck = String(lowered[lowered.index(after: apoIndex)...])
       } else {
           wordToCheck = lowered
       }
       let userDict = UserDictionary.shared
       if userDict.allLearnedWords[wordToCheck] != nil {
           return nil  // User-learned word: no correction needed
       }

       // Pass 2: trie spell check
       return aospTrieEngine.spellCheck(word)
   }
   ```

5. Update `injectUserWord(_ word:)`:
   - Change to no-op or keep calling `aospTrieEngine.injectUserWord(word)` (which is also no-op).
   - The two-pass lookup in spellCheck() now handles user words.
