```swift
import XCTest
@testable import DictusCore

final class FrequencyDictionaryTests: XCTestCase {

    func testRankReturnsCorrectValueForKnownWord() {
        var dict = FrequencyDictionary()
        let json = #"{"de": 1, "la": 2, "le": 3, "bonjour": 500}"#
        dict.load(from: json.data(using: .utf8)!)
        XCTAssertEqual(dict.rank(of: "de"), 1)
        XCTAssertEqual(dict.rank(of: "bonjour"), 500)
    }

    func testRankReturnsIntMaxForUnknownWord() {
        var dict = FrequencyDictionary()
        let json = #"{"de": 1}"#
        dict.load(from: json.data(using: .utf8)!)
        XCTAssertEqual(dict.rank(of: "xylophone"), Int.max)
    }

    func testRankIsCaseInsensitive() {
        var dict = FrequencyDictionary()
        let json = #"{"bonjour": 42}"#
        dict.load(from: json.data(using: .utf8)!)
        XCTAssertEqual(dict.rank(of: "Bonjour"), 42)
        XCTAssertEqual(dict.rank(of: "BONJOUR"), 42)
    }

    func testLoadFromInvalidDataProducesEmptyDict() {
        var dict = FrequencyDictionary()
        dict.load(from: "not json".data(using: .utf8)!)
        XCTAssertEqual(dict.rank(of: "de"), Int.max)
    }

    func testCommonWordsRankHigherThanUncommon() {
        var dict = FrequencyDictionary()
        let json = #"{"de": 1, "la": 2, "anticonstitutionnellement": 9999}"#
        dict.load(from: json.data(using: .utf8)!)
        XCTAssertTrue(dict.rank(of: "de") < dict.rank(of: "anticonstitutionnellement"))
    }

    func testLoadFromFixtureFile() {
        guard let url = Bundle.module.url(forResource: "fr_frequency_test", withExtension: "json", subdirectory: "Fixtures"),
              let data = try? Data(contentsOf: url) else {
            XCTFail("Test fixture fr_frequency_test.json not found in test bundle")
            return
        }
        var dict = FrequencyDictionary()
        dict.load(from: data)
        XCTAssertEqual(dict.rank(of: "de"), 1)
        XCTAssertTrue(dict.rank(of: "le") < dict.rank(of: "anticonstitutionnellement"))
    }
}
```

Create `DictusKeyboard/TextPrediction/` directory with 2 files:

**TextPredictionEngine.swift:**
- `import UIKit` and `import DictusCore` (for FrequencyDictionary and AccentedCharacters)
- Class (not struct -- holds UITextChecker instance which is a reference type)
- Properties: `private let textChecker = UITextChecker()`, `private var frequencyDict = FrequencyDictionary()`, `private var language: String = "fr"`
- `func setLanguage(_ lang: String)` -- updates language, reloads frequency dict via `frequencyDict.load(language: lang)`. For UITextChecker language param, use "fr" for French, "en" for English (verify against UITextChecker.availableLanguages at init and log if not found).
- `func suggestions(for partialWord: String) -> [String]` -- returns up to 3 ranked word completions:
  1. Call `textChecker.completions(forPartialWordRange:in:language:)` with the full partial word as the string and range covering all of it
  2. Sort results by `frequencyDict.rank(of:)` ascending (lower rank = more common = first)
  3. Return `Array(ranked.prefix(3))`
  4. If completions is empty or nil, return empty array
- `func spellCheck(_ word: String) -> String?` -- returns the best correction for a misspelled word, or nil if correctly spelled:
  1. Call `textChecker.rangeOfMisspelledWord(in:range:startingAt:wrap:language:)`
  2. If `misspelled.location == NSNotFound`, return nil (word is correct)
  3. Call `textChecker.guesses(forWordRange:in:language:)`
  4. Return the first guess ranked by frequency dict (not just `.first` -- re-rank guesses too)
- `func accentSuggestions(for partialWord: String) -> [String]?` -- returns accent variants for single-char vowel input:
  1. Guard `partialWord.count == 1` else return nil
  2. Look up `AccentedCharacters.mappings[partialWord.lowercased()]`
  3. If found, build array: `[partialWord] + variants.prefix(2)` (max 3 slots)
  4. Preserve case: if input is uppercase, uppercase the variants too
  5. Return the 3-element array, or nil if not a vowel

**SuggestionState.swift:**
- `import UIKit` and `import DictusCore`
- `enum SuggestionMode { case idle, completions, accents }`
- `class SuggestionState: ObservableObject` with:
  - `@Published var suggestions: [String] = []` (the 3 slots, can be fewer)
  - `@Published var mode: SuggestionMode = .idle`
  - `@Published var currentWord: String = ""` (the word being completed, needed for replacement)
  - `private let engine = TextPredictionEngine()`
  - `var autocorrectEnabled: Bool` -- reads from AppGroup.defaults SharedKeys.autocorrectEnabled
  - `var lastAutocorrect: AutocorrectState?` -- tracks the last autocorrection for undo
  - `func update(proxy: UITextDocumentProxy)` -- called on each keystroke:
    1. Extract current partial word from `proxy.documentContextBeforeInput` using `String.enumerateSubstrings(options: .byWords)` to find the last word. Only if text does NOT end with space/newline.
    2. If partial word is a single accentable vowel (first char after space), switch to accent mode via `engine.accentSuggestions(for:)`
    3. Otherwise, get completions via `engine.suggestions(for:)`
    4. Update `suggestions`, `mode`, `currentWord`
    5. If no partial word (after space, empty field), set mode to `.idle`, clear suggestions
  - `func performSpellCheck(_ word: String) -> String?` -- delegates to engine.spellCheck (needed by Plan 02 KeyboardView)
  - `func clear()` -- resets to idle state
  - `func setLanguage(_ lang: String)` -- forwards to engine
- `struct AutocorrectState` with: `originalWord: String`, `correctedWord: String`, `insertedSpace: Bool`
