public struct ModelInfo: Identifiable {
    public var id: String { identifier }
    public let identifier: String
    public let displayName: String
    public let engine: SpeechEngine
    public let sizeBytes: Int64
    public let sizeLabel: String
    public let accuracyScore: Double    // 0.0-1.0 for gauge
    public let speedScore: Double       // 0.0-1.0 for gauge
    public let description: String      // short French description
    public let visibility: CatalogVisibility

    public static let all: [ModelInfo]                    // only .available
    public static let allIncludingDeprecated: [ModelInfo]  // all models
    public static func forIdentifier(_ id: String) -> ModelInfo?
}

public enum SpeechEngine: String, Codable {
    case whisperKit = "WK"
    case parakeet = "PK"
    var displayName: String  // "WhisperKit" or "Parakeet"
}

public enum CatalogVisibility { case available, deprecated }
